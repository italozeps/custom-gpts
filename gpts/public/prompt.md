// Funkcija PDF nosaukuma iegūšanai
async function getPdfTitle(url) {
  try {
    const response = await fetch(url);
    if (!response.ok) throw new Error('Nav pieejams');
    
    const arrayBuffer = await response.arrayBuffer();
    const uint8Array = new Uint8Array(arrayBuffer);
    
    // Konvertē uz tekstu
    let text = '';
    for (let i = 0; i < Math.min(uint8Array.length, 10000); i++) {
      text += String.fromCharCode(uint8Array[i]);
    }
    
    // Meklē nosaukumu dažādos veidos
    let title = null;
    
    // 1. Meklē /Title tagā
    const titleMatch = text.match(/\/Title\s*\(([^)]+)\)/);
    if (titleMatch) {
      title = titleMatch[1];
    }
    
    // 2. Meklē starp stream objektiem
    if (!title) {
      const streamMatch = text.match(/stream[\s\S]*?BT[\s\S]*?\((.*?)\)\s*Tj/);
      if (streamMatch) {
        title = streamMatch[1];
      }
    }
    
    // 3. Meklē jebkuru tekstu iekavās pēc stream
    if (!title) {
      const anyTextMatch = text.match(/\(([A-Za-z0-9\s:',.\-?!]{10,100})\)/);
      if (anyTextMatch) {
        title = anyTextMatch[1];
      }
    }
    
    if (title) {
      // Attīra nosaukumu
      title = title.replace(/\\[rnt()]/g, ' ').trim();
      title = title.substring(0, 150); // Ierobežo garumu
      return title;
    }
    
    return "Nosaukums nav pieejams";
  } catch (error) {
    return "Kļūda: " + error.message;
  }
}

}

async function loadPdfJs() {
  if (typeof pdfjsLib !== 'undefined') {
    return; // Jau ielādēta
  }
  
  return new Promise((resolve, reject) => {
    const script = document.createElement('script');
    script.src = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.min.js';
    script.onload = () => {
      pdfjsLib.GlobalWorkerOptions.workerSrc = 
        'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js';
      resolve();
    };
    script.onerror = reject;
    document.head.appendChild(script);
  });
}

// Iegūst PDF nosaukumu izmantojot PDF.js
async function getPdfTitle(url) {
  try {
    const loadingTask = pdfjsLib.getDocument(url);
    const pdf = await loadingTask.promise;
    
    // Iegūst metadatus
    const metadata = await pdf.getMetadata();
    
    // Mēģina iegūt nosaukumu no metadata
    if (metadata.info && metadata.info.Title) {
      return metadata.info.Title;
    }
    
    // Ja nav metadata, ielādē pirmo lapu un mēģina iegūt tekstu
    const page = await pdf.getPage(1);
    const textContent = await page.getTextContent();
    
    // Iegūst pirmo līniju kā nosaukumu
    if (textContent.items.length > 0) {
      let firstLine = '';
      for (let i = 0; i < Math.min(5, textContent.items.length); i++) {
        const text = textContent.items[i].str.trim();
        if (text.length > 10) {
          firstLine = text;
          break;
        }
      }
      if (firstLine) {
        return firstLine.substring(0, 150);
      }
    }
    
    return "Nosaukums nav pieejams";
  } catch (error) {
    console.error(`Kļūda apstrādājot ${url}:`, error.message);
    return "Kļūda ielādējot";
  }
}
